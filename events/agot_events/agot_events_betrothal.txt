namespace = agot_betrothal_on_action_events

agot_betrothal_on_action_events.0001 = { # clear agot_delayed_wedding flag
	hidden = yes

	trigger = {
		OR = {
			has_character_flag = agot_delayed_wedding
			scope:second ?= { has_character_flag = agot_delayed_wedding }
		}
	}

	immediate = {
		if = {
			limit = { has_character_flag = agot_delayed_wedding }
			remove_character_flag = agot_delayed_wedding
		}
		if = {
			limit = { scope:second ?= { has_character_flag = agot_delayed_wedding } }
			scope:second = { remove_character_flag = agot_delayed_wedding }
		}
	}
}
