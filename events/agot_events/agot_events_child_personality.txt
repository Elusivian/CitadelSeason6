# Authoritative, Rude, inquisitive
agot_child_personality_events.2010 = {
	type = character_event
	title = agot_child_personality_events.2010.t
	desc = {
		first_valid = {
			triggered_desc = {
				trigger = {
					exists = scope:guardian
				}
				desc = agot_child_personality_events.2010.guardian
			}
			desc = agot_child_personality_events.2010.no_guardian
		}
	}
	theme = education
	left_portrait = {
		character = root
		animation = personality_rational
	}
	right_portrait = {
		character = scope:guardian
		animation = personality_callous
	}

	trigger = {
		is_available_child_allow_travel = yes
		NAND = {
			OR = {
				has_trait = authoritative
				has_trait = humble
			}
			OR = {
				has_trait = rude
				has_trait = compassionate
				has_trait = humble
			}
			OR = {
				has_trait = inquisitive
				has_trait = zealous
			}
		}
		OR = {
			AND = {
				is_ai = yes
				OR = {
					NOT = { exists = scope:guardian}
					scope:guardian = { is_ai = yes }
				}
			}
			is_landed = yes
			scope:guardian = { is_landed = yes }
		}
	}

	weight_multiplier = {
		base = 1
		modifier = {
			add = 2
			any_relation = {
				type = guardian
				OR = {
					has_trait = authoritative
					has_trait = rude
					has_trait = inquisitive
				}
			}
		}
	}

	immediate = {
		save_scope_as = child
		if = {
			limit = {
				any_relation = {
					type = guardian
					is_available = yes
				}
			}
			random_relation = {
				type = guardian
				save_scope_as = guardian
			}
		}
	}

	option = {
		name = agot_child_personality_events.2010.a
		trigger = {
			NOR = {
				has_trait = authoritative
				has_trait = humble
			}
		}
		ai_chance = {
			base = 1
			modifier = {
				add = 4
				any_relation = {
					type = guardian
					has_trait = authoritative
				}
			}
			modifier = {
				add = -0.5
				any_relation = {
					type = guardian
					has_trait = humble
				}
			}
		}
		add_trait = authoritative
		if = {
			limit = {
				is_ai = yes
			}
			random_relation = {
				type = guardian
				trigger_event = agot_child_personality_events.2011
			}
		}
	}

	option = {
		name = agot_child_personality_events.2010.b
		trigger = {
			NOR = {
				has_trait = humble
				has_trait = rude
				has_trait = compassionate
			}
		}
		ai_chance = {
			base = 1
			modifier = {
				add = 4
				any_relation = {
					type = guardian
					has_trait = rude
				}
			}
			modifier = {
				add = -0.5
				any_relation = {
					type = guardian
					or = {
						has_trait = diligent
						has_trait = compassionate
					}
				}
			}
		}
		add_trait = rude
		if = {
			limit = {
				is_ai = yes
			}
			random_relation = {
				type = guardian
				trigger_event = agot_child_personality_events.2012
			}
		}
	}

	option = {
		name = agot_child_personality_events.2010.c
		trigger = {
			NOR = {
				has_trait = inquisitive
				has_trait = zealous
			}
		}
		ai_chance = {
			base = 1
			modifier = {
				add = 4
				any_relation = {
					type = guardian
					OR = {
						has_trait = inquisitive
					}
				}
			}
			modifier = {
				add = -0.5
				any_relation = {
					type = guardian
					has_trait = zealous
				}
			}
		}
		add_trait = inquisitive
		if = {
			limit = {
				is_ai = yes
			}
			random_relation = {
				type = guardian
				trigger_event = agot_child_personality_events.2013
			}
		}
	}
}

# Child gains Authoritative
agot_child_personality_events.2011 = {
	type = character_event
	title = agot_child_personality_events.2011.t
	desc = {
		desc = agot_child_personality_events.201.desc
		desc = agot_child_personality_events.201.authoritative
	}
	theme = education

	left_portrait = {
		character = scope:child
		animation = personality_arrogant
	}
	right_portrait = {
		character = root
		animation = worry
	}

	trigger = {
		scope:child = {
			NAND = {
				OR = {
					has_trait = rude
					has_trait = compassionate
					has_trait = humble
				}
				OR = {
					has_trait = inquisitive
					has_trait = zealous
				}
			}
		}
	}

	option = {
		name = agot_child_personality_events.201.confirm_authoritative
		custom_tooltip = child_personality.keeps_authoritative_tt
	}
	option = {
		name = agot_child_personality_events.201.rude_change
		trigger = {
			scope:child = {
				NOR = {
					has_trait = rude
					has_trait = humble
					has_trait = compassionate
				}
			}
		}
		scope:child = {
			if = {
				limit = {
					NOT = { has_trait = rude }
				}
				add_trait = rude
			}
		}
		ai_chance = {
			base = 0
		}
		stress_impact = {
			base = child_trait_change_stress_penalty
		}
		scope:child = {
			remove_trait = authoritative
		}
	}
	option = {
		name = agot_child_personality_events.201.inquisitive_change
		trigger = {
			scope:child = {
				NOR = {
					has_trait = zealous
					has_trait = inquisitive
				}
			}
		}
		scope:child = {
			if = {
				limit = {
					NOT = { has_trait = inquisitive }
				}
				add_trait = inquisitive
			}
		}
		ai_chance = {
			base = 0
		}
		stress_impact = {
			base = child_trait_change_stress_penalty
		}
		scope:child = {
			remove_trait = authoritative
		}
	}
}
