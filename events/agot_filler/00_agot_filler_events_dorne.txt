agot_filler_dorne.0001 = {
	type = character_event
	title = agot_filler_dorne.0001.t
	desc = agot_filler_dorne.0001.desc

	override_background = {
		reference = throne_room_mediterranean
	}

	theme = seduction

	left_portrait = {
		character = root
		animation = throne_room_conversation_2
	}

	right_portrait = {
		character = scope:agot_filler_dorne_dueling_nobles_paramour
		animation = flirtation
	}

	trigger = {
		is_available_adult = yes
		is_landed = yes
		any_held_county = {
			title_province = { geographical_region = world_westeros_dorne }
		}
		NOT = {
			has_character_flag = agot_filler_dorne_dueling_nobles_paramour
		}
	}

	immediate = {
		play_music_cue = mx_cue_seduction
		add_character_flag = {
			flag = agot_filler_dorne_dueling_nobles_paramour
			#years = 25
		}
		create_character = {
			location = root.capital_province
			culture = root.culture
			faith = root.faith
			gender = female
			trait = wild_oat
			trait = education_intrigue_2
			trait = gregarious
			trait = lustful
			trait = ambitious
			trait = beauty_good_3
			trait = physique_good_3
			save_scope_as = agot_filler_dorne_dueling_nobles_paramour
		}
		save_scope_as = agot_filler_dorne_dueling_nobles_paramour_bastard_father
	}

	weight_multiplier = {
		base = 1
		modifier = {
			add = 0.5
			has_lifestyle = intrigue_lifestyle
		}
		modifier = {
			add = 0.5
			has_trait = lustful
		}
	}

	#Oh to live and love.... let them fight
	option = {
	   name = agot_filler_dorne.0001.a
	   custom_tooltip = agot_filler_dorne.0001.a_tt
	   hidden_effect = {
			scope:agot_filler_dorne_dueling_nobles_paramour = {
				death = {
					death_reason = death_disappearance
				}
			}
	   }
	   stress_impact = {
			calm = minor_stress_gain
			callous = minor_stress_loss
			lustful = minor_stress_loss
			greedy = minor_stress_impact_gain
			generous = medium_stress_impact_loss
		}
	}

	#No dueling over women allowed!
	option = {
		name = agot_filler_dorne.0001.b
		custom_tooltip = agot_filler_dorne.0001.b_tt
		hidden_effect = {
			scope:agot_filler_dorne_dueling_nobles_paramour = {
				death = {
					death_reason = death_disappearance
				}
			}
	   }
		stress_impact = {
			calm = minor_stress_loss
			callous = minor_stress_gain
			just = minor_stress_impact_loss
			arbitrary = minor_stress_impact_gain
		}
	}

	#take the paramour for yourself, but you get a bastard 9 months later.
	option = {
		name = agot_filler_dorne.0001.c
		custom_tooltip = agot_filler_dorne.0001.c_tt
		hidden_effect = {
			scope:agot_filler_dorne_dueling_nobles_paramour = {
				death = {
					death_reason = death_disappearance
				}
			}
	   }
		trigger = {
			OR = {
				has_trait = lustful
				has_trait = seducer
			}
		}
		show_as_unavailable = {
			NOR = {
				has_trait = lustful
				has_trait = seducer
			}
		}
		trait = lustful
		trait = seducer

		hidden_effect = {
			if = {
				limit = {
					is_male = yes
				}
				random = {
					chance = 88
					trigger_event = {
						id = agot_filler_dorne.0002
						days = { 274 280 }
					}
				}
			}
		}
	}

}
