							modifier = agot_filler_hurricane_wrecked_stepstones

					}

			}
		}
		random = {
			chance = 50
			show_as_tooltip = {
				custom_tooltip = agot_filler_stepstones.0001.a.tt.b
			}
		}
	}

	#Not my problem.
	option = {
		name = agot_filler_stepstones.0001.b

			limit = { has_trait = just  }

				chance = 75

			just = massive_stress_impact_gain
