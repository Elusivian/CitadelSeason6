agot_filler_war.100 = { # Our coffers are empty. This war has been drying them up!
	type = character_event
	title = agot_filler_war.100.t
	desc = agot_filler_war.100.desc
	theme = realm
	override_background = {
		reference = throne_room_west
	}

	left_portrait = {
		character = root
		animation = idle
	}

	right_portrait = {
		character = cp:councillor_steward
		animation = personality_rational
	}

	trigger = {
		exists = scope:war
		is_at_war = yes
		is_available_allow_travelling = yes
		NOT = {
			has_character_flag = expensive_war_agot
		}
		OR = {
			days_of_continuous_war > 180
			gold < 50
		}
	}

	immediate = {
		add_character_flag = expensive_war_agot
	}

	option = {	# Sue for peace
		name = agot_filler_war.100.a
		custom_tooltip = agot_filler_war.100.a_tt
		if = {
			limit = {
				has_trait = ambitious
			}
			add_stress = 50
			scope:war = {
				end_war = white_peace
			}
			random_list = {
				50 = {
					add_stress = 40
					modifier = {
						add = 50
						has_trait = wrathful
					}
				}
				20 = {
					add_stress = 40
					modifier = {
						add = 50
						has_trait = callous
					}
				}
			}
		}
		else = {
			add_stress = 20
		}
	}

	option = {	# We shall ask the Iron Bank for a loan.
		name = agot_filler_war.100.b
		custom_tooltip = agot_filler_war.100.b_tt
		trigger = {
			loaned_from_iron_bank_trigger = no
		}
		trigger_event = {
			id = agot_filler_war.101
			months = { 1 3 }
		}
	}

	option = {	# We shall do nothing. We will endure until victory comes.
		name = agot_filler_war.100.c
		custom_tooltip = agot_filler_war.100.c_tt
		if = {
			limit = {
				OR = {
					has_trait = callous
					has_trait = sadistic
				}
			}
			add_stress = -25
		}
		add_character_modifier = {
			modifier = badly_supplied_armies_modifier
			years = 5
		}
	}
}

		character = root
		triggered_animation = {
			trigger = {
				has_trait = craven
			}
			animation = fear

	}

		location = {
			geographical_region = world_westeros
		}

			is_army_in_combat = yes
		}
	}

	immediate = {
		add_character_flag = down_the_slopes_of_death

				desc = agot_filler_war.200.list_1 # I am the last of my line.. If I should fall, let it be worthy of remembrance, upon my loyal steed and my lance gripped firm in my hand.

					add = {
						value = 1

							limit = {
								OR = {
									has_trait = brave

									has_trait = ambitious
								}
							}
							add = 100
						}
					}
				}
			}

				desc = agot_filler_war.200.list_2 # I can't hold myself together any longer, I have not a warrior's soul.

						if = {
							limit = {
								has_trait = craven
							}

						}

		}
	}

	option = {	# They shall sing songs about my deeds...
		name = agot_filler_war.200.a
		custom_tooltip = agot_filler_war.200.a_tt
		trigger = {
			NOT = {

	}

	option = {	# Chaos reigns around me...
		name = agot_filler_war.200.b
		custom_tooltip = agot_filler_war.200.b_tt

# The Hanged Man
##########################################

agot_filler_war.400 = { #
	type = character_event
	title = agot_filler_war.400.t
	desc = agot_filler_war.400.desc
	theme = war
	override_background = {
		reference = burning_building
	}


	left_portrait = {
		character = root
		animation = disbelief
		triggered_animation = {
			trigger = {
				OR = {
					has_trait = sadistic
					has_trait = callous
				}
			}
			animation = dismissal
		}
	}

	right_portrait = {
		character = scope:soldier
	}

	lower_center_portrait = {
		character = scope:hanged_baby
	}

	trigger = {
		location = {
			geographical_region = world_westeros_the_riverlands
		}
		is_at_war = yes
		is_commanding_army = yes
		NOT = {
			exists = global_var:baby_guts
		}
		root.commanding_army = {
			is_army_in_combat = no
		}
	}

	immediate = {
		set_global_variable = {
			name = baby_guts
			value = yes
			days = 10000
		}

		create_character = {
			age = 0
			dynasty = none
			location = this.location
			faith = root.faith
			culture = root.culture
			gender_female_chance = root_soldier_female_chance
			save_scope_as = hanged_baby
			trait = strong
			trait = physique_good_2
		}

		if = {
			limit = {
				any_courtier = {
					location = root.location
					is_available_adult = yes
					is_knight_of = root
					ai_compassion < 0 #cool killing the baby
				}
			}
			random_courtier = {
				limit = {
					is_available_adult = yes
					is_knight_of = root
					ai_compassion < 0 #cool killing the baby
				}
				save_scope_as = soldier
			}
		}
		else = {
			create_character = {
				template = new_commander_character
				dynasty = none
				employer = root
				gender = male
				faith = root.faith
				culture = root.culture
				save_scope_as = soldier
			}
			scope:soldier = {
				if = {
					limit = {
						has_trait = compassionate
					}
					remove_trait = compassionate
				}
				if = {
					limit = {
						NOT = { has_trait = sadistic }
					}
					add_trait = callous
				}
			}

		}
	}

	option = {
		name = agot_filler_war.400.a

		add_prestige = minor_prestige_loss

		add_courtier ?= scope:hanged_baby

		ai_chance = {
			base = 40
		}
	}

	option = {
		name = agot_filler_war.400.b

		trigger = {
			is_male = yes
			NOT = {
				has_trait = chaste
				has_trait = celibate
				has_trait = eunuch
			}
		}
		add_courtier ?= scope:hanged_baby
		scope:hanged_baby = {
			set_house = root.house
			add_trait = bastard
			give_nickname = nick_rivers
			set_father = root
		}
		stress_impact = {
			compassionate = massive_stress_impact_loss
			paranoid = medium_stress_impact_gain
			callous = major_stress_impact_gain
			sadistic = major_stress_impact_gain
		}

		ai_chance = {
			base = 20
			modifier = {
				factor = 0
				OR = {
					has_trait = paranoid
					has_trait = arrogant
					has_trait = callous
					has_trait = sadistic
					ai_honor >= high_positive_honor #Won't sully their honor taking a random bastard
				}
			}
		}
	}

	option = {
		name = agot_filler_war.400.c
		scope:hanged_baby = {
			death = {
				death_reason = death_disappearance
			}
		}
		add_piety = major_piety_gain

		ai_chance = {
			base = 40
			modifier = { #Nice peoples don't let babies die
				factor = 0
				ai_compassion >= 20
			}
		}
	}

}
