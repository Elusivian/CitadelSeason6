agot_filler_reach.100 = { # A few lords have reported arson attacks. Not anywhere important, specifically in the flowerbeds in the garden. That normally wouldn’t be a problem, but flowerbeds make up rather a lot of the decorations.
	type = character_event
	title = agot_filler_reach.100.t
	desc = agot_filler_reach.100.desc
	theme = realm
	override_background = {
		reference = farmland
	}

	left_portrait = {
		character = root
		animation = shock
	}

	right_portrait = {
		character = cp:councillor_marshal
		animation = personality_bold
	}

	trigger = {
		exists = cp:councillor_marshal
		exists = cp:councillor_steward
		exists = cp:councillor_spymaster
		exists = location
		location = { geographical_region = world_westeros_the_reach }
		is_at_war = no
		is_available_adult = yes
		is_imprisoned = no
		NOT = {
			has_character_flag = arson_reach
		}
	}

	immediate = {	#for redundance
		add_character_flag = {
			flag = arson_reach
			years = 5
		}
		add_character_modifier = {
			modifier = burned_lands_modifier
			years = 5
		}
		cp:councillor_marshal = { save_scope_as = marshal }
		cp:councillor_steward = { save_scope_as = steward }
		cp:councillor_spymaster = { save_scope_as = spymaster }
	}

	option = {	#Let's see what we can do, after the fires go down.
		name = agot_filler_reach.100.a
		custom_tooltip = agot_filler_reach.100.a_tt
		remove_short_term_gold = 100
		trigger_event = {
			id = agot_filler_reach.101
			days = { 3 4 }
		}
	}
}

				geographical_region = world_westeros_the_reach
			}
		}
		NOT = {
			has_character_flag = agot_filler_bee_sting
		}
	}
	immediate = {	#for redundance
		add_character_flag = {
			flag = agot_filler_bee_sting
			years = 5
		}

		add_trait = wounded_1
	}

	option = {	# NOT THE BEES!!
		name = agot_filler_reach.200.a
		add_prestige = minor_prestige_loss

		ai_chance = {
			base = 1
		}
	}

	option = {	# treat the sting
		name = agot_filler_reach.200.b
		trigger = {
			has_trait = lifestyle_herbalist
		}
		remove_trait = wounded_1
		add_prestige = minor_prestige_gain

		ai_chance = {
			base = 100
		}
	}

	option = {	#burn the garden
		name = agot_filler_reach.200.c
		trigger = {
			NOT = {
				has_trait = calm
			}
		}
		if = {
			limit = {
				number_of_personality_traits < personality_trait_limit
				NOT = { has_trait = wrathful }
			}
			add_trait = wrathful
		}
		add_prestige = medium_prestige_loss

		ai_chance = {
			base = 0
		}
	}
}
