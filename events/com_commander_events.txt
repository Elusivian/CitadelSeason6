fow_trait.0001 = {

	hidden = yes

    trigger = {
        exists = scope:army.army_commander
            scope:army.army_commander = {
                has_trait = seasoned_rover
            }
        }

    immediate = {
        scope:army.army_commander = {
            save_scope_as = army_leader
        }
        if = {
            limit = {
                OR = {
                    scope:army_leader.location.county.holder = {
                        is_vassal_of = root
                    }
                scope:army_leader.location.county.holder = root
                }
            }
            scope:army_leader = {
                if = {
                    limit = {
                        has_trait_xp = {
                            trait = seasoned_rover
                                    value >= 100
                        }
                    }
                    add_character_modifier = {
                        modifier = seasoned_rover_neg_modifier_3
                    }
                    remove_character_modifier = seasoned_rover_neg_modifier
                    remove_character_modifier = seasoned_rover_neg_modifier_1
                    remove_character_modifier = seasoned_rover_neg_modifier_2
                }
                else_if = {
                    limit = {
                        has_trait_xp = {
                            trait = seasoned_rover
                                    value >= 60
                        }
                    }
                    add_character_modifier = {
                        modifier = seasoned_rover_neg_modifier_2
                    }
                    remove_character_modifier = seasoned_rover_neg_modifier
                    remove_character_modifier = seasoned_rover_neg_modifier_1
                    remove_character_modifier = seasoned_rover_neg_modifier_3
                }
                else_if = {
                    limit = {
                        has_trait_xp = {
                            trait = seasoned_rover
                                    value >= 35
                        }
                    }
                    add_character_modifier = {
                        modifier = seasoned_rover_neg_modifier_1
                    }
                    remove_character_modifier = seasoned_rover_neg_modifier
                    remove_character_modifier = seasoned_rover_neg_modifier_1
                    remove_character_modifier = seasoned_rover_neg_modifier_2
                }
                else_if = {
                    limit = {
                        has_trait_xp = {
                            trait = seasoned_rover
                                    value >= 0
                        }
                    }
                    add_character_modifier = {
                        modifier = seasoned_rover_neg_modifier
                    }
                    remove_character_modifier = seasoned_rover_neg_modifier_1
                    remove_character_modifier = seasoned_rover_neg_modifier_2
                    remove_character_modifier = seasoned_rover_neg_modifier_3

                remove_character_modifier = seasoned_rover_pos_modifier
                remove_character_modifier = seasoned_rover_pos_modifier_1
                remove_character_modifier = seasoned_rover_pos_modifier_2
                remove_character_modifier = seasoned_rover_pos_modifier_3
            }
        }
        else = {
            scope:army_leader = {
                if = {
                    limit = {
                        has_trait_xp = {
                            trait = seasoned_rover
                                    value >= 100
                        }
                    }
                    add_character_modifier = {
                        modifier = seasoned_rover_pos_modifier_3
                    }
                    remove_character_modifier = seasoned_rover_pos_modifier
                    remove_character_modifier = seasoned_rover_pos_modifier_1
                    remove_character_modifier = seasoned_rover_pos_modifier_2
                }
                else_if = {
                    limit = {
                        has_trait_xp = {
                            trait = seasoned_rover
                                    value >= 60
                        }
                    }
                    add_character_modifier = {
                        modifier = seasoned_rover_pos_modifier_2
                    }
                    remove_character_modifier = seasoned_rover_pos_modifier
                    remove_character_modifier = seasoned_rover_pos_modifier_1
                    remove_character_modifier = seasoned_rover_pos_modifier_3
                }
                else_if = {
                    limit = {
                        has_trait_xp = {
                            trait = seasoned_rover
                                    value >= 35
                        }
                    }
                    add_character_modifier = {

fow_trait.0004 = {

	hidden = yes    

    trigger = {
        exists = scope:army.army_commander
            scope:army.army_commander = {
                has_trait = infiltration_specialist
            }
        }

    immediate = {
        scope:army.army_commander = {
            save_scope_as = army_leader
        }
        if = {
            limit = {
                OR = {
                    scope:army_leader.location.county.holder = {
                        OR = {
                            is_at_war_with = scope:army_leader
                            scope:army_leader = {
                                any_liege_or_above = {
                                    is_at_war_with = scope:army_leader.location.county.holder
                                }
                            }
                        }
                    }
                    scope:army_leader.location.county.holder = {
                        any_liege_or_above = {
                            OR = {
                                is_at_war_with = scope:army_leader
                                scope:army_leader = {
                                    any_liege_or_above = {
                                        is_at_war_with = scope:army_leader
                                    }
                                }
                            }
                        }
                    }
                }
                scope:army_leader.location.county.holder = {
                    intrigue < army_leader.intrigue
                }
            }
            scope:army_leader = {
                add_character_modifier = {
                    modifier = infiltration_specialist_modifier
                }
            }
        }
        else = {
            scope:army_leader = {
                remove_character_modifier = infiltration_specialist_modifier
            }
        }
    }
}

fow_trait.00071 = {

	hidden = yes

    trigger = {
        exists = scope:army.army_commander
            scope:army.army_commander = {
                has_trait = focus_specialist
            }
        }

    immediate = {
        scope:army.army_commander = {
            if = {
                limit = {
                    is_at_war_with = this.location.county.holder
                    this.location.county = this.location.county.holder.primary_title.title_capital_county
                }
                add_character_modifier = focus_specialist_modifier
            }
            else = {
                remove_character_modifier = focus_specialist_modifier
            }
        }
    }
}
