agot_prowess_education_trigger = {
	OR = {
		OR = {
			can_be_warrior_trigger_no_dragon = { ARMY_OWNER = root }
			can_be_combatant_based_on_gender_trigger = { ARMY_OWNER = root }
		}
		AND = {
			exists = liege_or_court_owner
			OR = {
				liege_or_court_owner = {
					has_title = title:d_bear_island
				}
				liege_or_court_owner = {
					any_liege_or_above = {
						has_title = title:d_bear_island
					}
				}
			}
		}
		has_character_flag = tomboy
	}

	# OR = {
	# 	#If the child belongs to a knightly culture, then their martial education includes squirehood.
	# 	culture = { has_innovation = innovation_knighthood }
	# 	AND = {	#If the child's guardian belongs to a knightly culture, then they will naturally introduce them to the rigours of squiredom.
	# 		exists = scope:educator
	# 		scope:educator = {
	# 			culture = { has_innovation = innovation_knighthood }
	# 		}
	# 	}
	# }
	trigger_if = {
		limit = {
			exists = scope:educator
			exists = scope:educator.liege_or_court_owner
			exists = scope:educator.liege_or_court_owner.capital_county.title_province
		}
		NOT = { scope:educator.liege_or_court_owner.capital_county.title_province = { geographical_region = world_essos_free_cities } }
	}
	trigger_else_if = {
		limit = {
			exists = liege_or_court_owner
			exists = liege_or_court_owner.capital_county.title_province
		}
		NOT = { liege_or_court_owner.capital_county.title_province = { geographical_region = world_essos_free_cities } }
	}
	trigger_else = {
		NOT = { location = { geographical_region = world_essos_free_cities } }
	}
}
