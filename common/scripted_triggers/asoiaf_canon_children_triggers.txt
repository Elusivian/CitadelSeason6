	asoiaf_historical_trigger = yes

	scope:real_father = { has_inactive_trait = asoiaf_Baratheon_2_trait } #Robert I Baratheon
	scope:mother = {
		has_inactive_trait = asoiaf_Lannister_6_trait #Cersei Lannister (daughter of Tywin)

			num_of_relation_lover > 0 #if Cersei has a lover, but that's NOT Robert, she aborts the child (doesn't want to bear Robert's child while she has a lover)
			NOT = { has_relation_lover = scope:real_father }
		}
	}
}
