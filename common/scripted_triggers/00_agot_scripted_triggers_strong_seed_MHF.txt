agot_strong_seed_house_stability_trigger = {
	has_game_rule = agot_strong_seed_house_stability
	$PARENT$ = { save_temporary_scope_as = parent }
	OR = {
		AND = {
			is_female = yes
			scope:parent = {
				OR = {
					any_child = {
						count = 1
						is_female = yes
						dynasty = scope:parent.dynasty
						NOT = { has_trait = bastard }
					}
					any_child = {
						count = 2
						is_female = yes
						dynasty = scope:parent.dynasty
						has_trait = twin
						NOT = { has_trait = bastard }
					}
				}
			}
		}
		AND = {
			is_male = yes
			scope:parent = {
				OR = {
					any_child = {
						count = 1
						is_male = yes
						dynasty = scope:parent.dynasty
						NOT = { has_trait = bastard }
					}
					any_child = {
						count = 2
						is_male = yes
						dynasty = scope:parent.dynasty
						has_trait = twin
						NOT = { has_trait = bastard }
					}
				}
			}
		}
	}
}
