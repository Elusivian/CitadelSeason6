has_kingsblood = {
	OR = {
		highest_held_title_tier = tier_empire
		AND = {
			agot_is_independent_ruler = yes
			any_held_title = {
				tier = tier_kingdom
			}
		}
		any_close_family_member = {
			OR = {
				highest_held_title_tier = tier_empire
				AND = {
					agot_is_independent_ruler = yes
					any_held_title = {
						tier = tier_kingdom
					}
				}
			}
		}
	}
}

has_paramountblood = {
	OR = {
		has_kingsblood = yes
		highest_held_title_tier = tier_kingdom
		any_close_family_member = {
			highest_held_title_tier = tier_kingdom
		}
	}
}

has_dukesblood = {
	OR = {
		has_paramountblood = yes
		highest_held_title_tier = tier_duchy
		any_close_family_member = {
			highest_held_title_tier = tier_duchy
		}
	}
}

has_countsblood = {
	OR = {
		has_dukesblood = yes
		highest_held_title_tier = tier_county
		any_close_family_member = {
			highest_held_title_tier = tier_county
		}
	}
}
