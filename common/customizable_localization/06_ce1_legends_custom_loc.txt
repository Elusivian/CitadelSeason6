GetRandomExteriorDetail = { #Remember to use the 1st person present in the loc, feel free to expand
	type = character
	random_valid = yes

	text = { # You see a small bird among the trees
		trigger = {
			location = {
				OR = {
					#AGOT Modified
					#terrain = forest
					#terrain = jungle
					agot_is_forest_terrain = yes
					agot_is_jungle_terrain = yes
					agot_is_cloudforest_terrain = yes
				}
			}
		}
		localization_key = GetRandomExteriorDetail_small_bird
	}
	text = { # You hear a bird
		trigger = {
			location = {
				OR = {
					#AGOT Modified
					#terrain = forest
					#terrain = jungle
					#terrain = plains
					#terrain = hills
					#terrain = mountains
					agot_is_forest_terrain = yes
					agot_is_jungle_terrain = yes
					agot_is_plains_terrain = yes
					agot_is_hills_terrain = yes
					agot_is_mountains_terrain = yes
					agot_is_cloudforest_terrain = yes
					agot_is_highlands_terrain = yes
				}
			}
		}
		localization_key = GetRandomExteriorDetail_birdsong
	}
	text = { # You find a wildflower
		trigger = {
			location = {
				OR = {
					#AGOT Modified
					#terrain = desert
					#terrain = desert_mountains
					#terrain = drylands
					#terrain = mountains
					agot_is_desert_terrain = yes
					agot_is_desert_mountains_terrain = yes
					agot_is_drylands_terrain = yes
					agot_is_mountains_terrain = yes
					agot_is_glacier_terrain = yes
					agot_is_canyon_terrain = yes
				}
			}
		}
		localization_key = GetRandomExteriorDetail_wildflower
	}
	text = { # You hear someone singing
		trigger = {
			location = {
				has_holding_type = castle_holding
			}
		}
		localization_key = GetRandomExteriorDetail_singing
	}
	text = { # You smell a banquet
		trigger = {
			location = {
				has_holding_type = castle_holding
			}
		}
		localization_key = GetRandomExteriorDetail_banquet
	}
