#SIGILS, TERRAIN, TITLES

GetCadetName = {
	type = character
	random_valid = yes

	text = {
		trigger = {
			NOT = {
				has_house_sigil_trigger = yes
				has_house_name_trigger = yes
			}
		} #Generic
		localization_key = GetCadetHouseName_A  #Prefix-Suffix
	}
	text = {
		trigger = { has_house_sigil_trigger = yes }
		localization_key = GetCadetHouseName_B  #Prefix-HouseSuffix (with house symbol)
	}
	text = {
		trigger = { has_house_name_trigger = yes }
		localization_key = GetCadetHouseName_C   #HousePrefix-Suffix
	}
	text = {
		trigger = { has_valyrian_house_name_trigger = yes }
		localization_key = GetCadetHouseName_D #HousePrefix-HouseSuffix ( Super Rare )
	}
}
