more_bookmarks_aegon_I_simple_canon_kids = {
	# by Freeholder

	on_setup = {
		debug_log = "Canon Kids initialized"
		story_owner = { add_character_modifier = simple_canon_kids_infertility }
		set_variable = {
			name	= aenys
			value	= flag:unborn
		}
		set_variable = {
			name	= maegor
			value	= flag:unborn
		}
		
	}
	on_end = {
	if = { limit = { story_owner = { is_alive = yes } }
	story_owner = { remove_character_modifier = simple_canon_kids_infertility } 
	}
	}
	on_owner_death = {
	if = { limit = { exists = story_owner.primary_heir }
	make_story_owner = story_owner.primary_heir 
	}
	else_if = { limit = { story_owner = { any_child = { is_male = yes is_alive = yes } } }
		random_child = { limit = { is_male = yes is_alive = yes } save_scope_as = new_owner }
		scope:new_owner = { add_character_modifier = simple_canon_kids_infertility }
		make_story_owner = scope:new_owner
	}
	else = {
	character:Targaryen_27.house.house_head = { add_character_modifier = simple_canon_kids_infertility }
	make_story_owner = character:Targaryen_27.house.house_head
	}

	}

	# General daily checks
	effect_group = {
		days = 7

		# 
		triggered_effect = {
			trigger = {
				current_date >= 8007.1.1
				var:aenys = flag:unborn
				NOT = { story_owner = { has_trait = pregnant } }
				NOT = { story_owner = { has_character_flag = canon_birth_cooldown } }
			}

			effect = {
			if = { limit = { story_owner = { is_male = yes } } 
		story_owner = { save_scope_as = dad  }
		story_owner.primary_spouse = { save_scope_as = mom } 
		}
		else = {
		story_owner =  { save_scope_as = mom }
		story_owner.primary_spouse = { save_scope_as = dad } 
		}
		if = { limit = { character:Targaryen_27 = { is_alive = yes } character:Targaryen_29 = { is_alive = yes } }
				character:Targaryen_27 = { save_scope_as = dad }
				character:Targaryen_29 = { save_scope_as = mom }
				}
				
				if = { limit = { exists = scope:mom 
								 exists = scope:dad
				} 
							random_list = {
											10 = { # Child is born
												
								
												
								
												
												
												set_variable = {
																name	= aenys
																value	= flag:born
												}
												simple_canon_kids_aenys_targaryen_birth_effect = { FATHER = scope:dad MOTHER = scope:mom }
												}
												
												
												
											
										
											200 = { # nothing happens
												
												
											}
							}
							}
			
			
			}
		}
	 
		triggered_effect = {
			trigger = {
				current_date >= 8012.1.1
				var:maegor = flag:unborn
				NOT = { story_owner = { has_trait = pregnant } }
				NOT = { story_owner = { has_character_flag = canon_birth_cooldown } }
			}

		effect = {
			if = { limit = { story_owner = { is_male = yes } } 
		story_owner = { save_scope_as = dad  }
		story_owner.primary_spouse = { save_scope_as = mom } 
		}
		else = {
		story_owner =  { save_scope_as = mom }
		story_owner.primary_spouse = { save_scope_as = dad } 
		}
		if = { limit = { character:Targaryen_27 = { is_alive = yes } character:Targaryen_28 = { is_alive = yes } }
				character:Targaryen_27 = { save_scope_as = dad }
				character:Targaryen_28 = { save_scope_as = mom }
