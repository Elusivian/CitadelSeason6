	if = {
		limit = { exists = $NUM$ }

			trigger = $NUM$
			0 = { save_scope_value_as = { name = vsr_flag value = flag:wolf } }
			1 = { save_scope_value_as = { name = vsr_flag value = flag:eagle } }
			2 = { save_scope_value_as = { name = vsr_flag value = flag:lizard } }
			3 = { save_scope_value_as = { name = vsr_flag value = flag:crow } }
			4 = { save_scope_value_as = { name = vsr_flag value = flag:dog } }
			5 = { save_scope_value_as = { name = vsr_flag value = flag:dragon } }
			6 = { save_scope_value_as = { name = vsr_flag value = flag:fish } }
			7 = { save_scope_value_as = { name = vsr_flag value = flag:lion } }
			8 = { save_scope_value_as = { name = vsr_flag value = flag:scorpion } }
			9 = { save_scope_value_as = { name = vsr_flag value = flag:rose } }
		}
	}
}
