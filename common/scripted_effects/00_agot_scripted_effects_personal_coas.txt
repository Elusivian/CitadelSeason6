agot_create_personal_coa = {
	save_scope_as = actor
	add_character_flag = has_personal_coa
	if = {
		limit = {
			scope:actor = { 
				is_playable_character = yes
			}
		}
		create_character = {
			name = "Dummy"
			gender = male
			employer = scope:actor
			trait = agot_dummy_trait
			random_traits = no
			faith = scope:actor.faith
			culture = scope:actor.culture
			dynasty = generate
			save_scope_as = recipient
		}
	}
	else_if = {
		limit = {
			scope:actor = {	
				is_playable_character = no 
			}
		}
		if = {
			limit = {
				character:Ruin_Empress = {
					is_alive = yes
				}
			}
			create_character = {
				name = "Dummy"
				gender = male
				employer = character:Ruin_Empress
				trait = agot_dummy_trait
				random_traits = no
				faith = scope:actor.faith
				culture = scope:actor.culture
				dynasty = generate
				save_scope_as = recipient
			}
		}
		else_if = {
			limit = {
				character:Ruin_Empress = {
					is_alive = no
				}
			}
			title:c_ruins.holder = {
				save_scope_as = ruins
			}
			create_character = {
				name = "Dummy"
				gender = male
				employer = scope:ruins
				trait = agot_dummy_trait
				random_traits = no
				faith = scope:actor.faith
				culture = scope:actor.culture
				dynasty = generate
				save_scope_as = recipient
			}
		}
	}
	scope:recipient = {
		set_global_variable = {
			name = coa_new_name
			value = scope:actor
		}
		scope:recipient.house = {
			set_house_name = coa_new_name
			# else_if... sibling count e.g. fourth child, third child, etc
			# NOTE TO SELF.. maybe use a random_list to select which option?
		}
		# scope:recipient.dynasty = {
			# add_dynasty_prestige_level = scope:actor.dynasty.dynasty_prestige_level
			# add_dynasty_prestige_level = -1
		# }
		remove_global_variable = coa_new_name
		set_variable = {
			name = my_coa_belongs_to
			value = scope:actor
		}
	}
	if = {
		limit = {
			scope:actor = {
				has_nickname_for_personal_coa_trigger = yes #Using this because black/white/red nickname triggers are including nicknames that has no personal coa attached to.
			}
		}
		agot_select_personal_coa_based_on_nickname_system = yes
	}
	else_if = {
		limit = {
			scope:actor = {
				OR = {
					has_faith = faith:rhllor
					has_faith = faith:red_god
				}
				has_specific_rhllor_personal_coa_trigger = yes
			}
		}
		agot_select_special_rhllor_personal_coa = yes
	}
	else_if = {
		limit = {
			scope:actor = {
				OR = {
					has_faith = faith:rhllor
					has_faith = faith:red_god
				}
				has_rhllor_personal_coa_trigger = yes
			}
		}
		agot_select_rhllor_personal_coa = yes
	}
	else_if = {
		limit = {
			scope:actor = {
				any_sibling = {
					count = 3
					AND = {
						sex_same_as = scope:actor
						age > scope:actor.age
					}
				}
			}
		}
		agot_select_personal_coa_based_on_birth_number = yes
	}
	else_if = {
		limit = { scope:actor = { has_character_flag = selecting_faction_coa } }
		if = {
			limit = {
				scope:actor = {
					OR = {
						AND = {
							is_character_targaryen_63 = yes
							is_married = yes
							exists = dynasty:dynn_Velaryon.dynasty_founder.house
							scope:actor.primary_spouse.house = dynasty:dynn_Velaryon.dynasty_founder.house
						}
						is_character_targaryen_65 = yes
					}
				}
			}
			if = {
				limit = {
					is_character_targaryen_63 = yes
					is_married = yes
					exists = dynasty:dynn_Velaryon.dynasty_founder.house
					scope:actor.primary_spouse.house = dynasty:dynn_Velaryon.dynasty_founder.house
				}
				scope:recipient.house = { set_coa = rhaenrya_personal_coa }
			}
			else_if = {
				limit = {
					is_character_targaryen_65 = yes
				}
				scope:recipient.house = { set_coa = aegon_ii_personal_arms }
			}
		}
		else = {
			random_list = {
				99 = {
					trigger = {
						scope:actor = {
							has_trait = dragonrider
							any_relation = {
								type = agot_dragon
								is_alive = yes
								can_have_dragon_color_coa = yes
							}
						}
					}
					scope:recipient = { assign_dragon_color_for_coa = yes }
					scope:actor = { agot_select_faction_coa = yes }
					scope:recipient.house = { generate_coa = agot_faction_coa }
				}
				1 = {
					scope:actor = { agot_select_faction_coa = yes }
					scope:recipient.house = { generate_coa = agot_faction_coa }
				}
			}
		}
	}
	else_if = {
		limit = {
			scope:actor = {
				is_lowborn = yes
			}
		}
		scope:recipient.house = { generate_coa = yes }
	}
	else = {
		scope:recipient.house = { set_coa = scope:actor.house }
	}
	scope:actor = {
		set_variable = {
			name = my_personal_coa
			value = scope:recipient
		}
		set_relation_agot_personal_coa_slave = scope:recipient
		if = {
			limit = {
				scope:actor = {
					is_ai = yes
				}
			}
			scope:recipient = { save_scope_as = exiled_dummy } # Send away the COA Dummy Immediately since the AI won't be editing their own COAs
			trigger_event = agot_events_personal_coat_of_arms.0001
		}
		else = {
			scope:recipient = { save_scope_as = exiled_dummy } # Send away the COA Dummy a few days after
			trigger_event = { id = agot_events_personal_coat_of_arms.0001 days = 1 }
		}
	}
}
agot_create_personal_coa_for_historical_ai = {
	$ACTOR$ = {
		save_scope_as = actor
		add_character_flag = has_personal_coa
		if = {
			limit = {
				character:Ruin_Empress = {
					is_alive = yes
				}
			}
			create_character = {
				name = "Dummy"
				gender = male
				employer = character:Ruin_Empress # AI Doesn't need to edit their COA, so send the dummy away immediately.
				trait = agot_dummy_trait
				faith = scope:actor.faith
				culture = scope:actor.culture
				dynasty = generate
				save_scope_as = recipient
			}
		}
		else_if = {
			limit = {
				character:Ruin_Empress = {
					is_alive = no
				}
			}
			title:c_ruins.holder = {
				save_scope_as = ruins
			}
			create_character = {
				name = "Dummy"
				gender = male
				employer = scope:ruins # AI Doesn't need to edit their COA, so send the dummy away immediately.
				trait = agot_dummy_trait
				faith = scope:actor.faith
				culture = scope:actor.culture
				dynasty = generate
				save_scope_as = recipient
			}
		}
		scope:recipient = {
			set_global_variable = {
				name = coa_new_name
				value = scope:actor
			}
			scope:recipient.house = {
				set_house_name = coa_new_name
				set_coa = $COA_KEY$
			}
			# scope:recipient.dynasty = {
				# add_dynasty_prestige_level = scope:actor.dynasty.dynasty_prestige_level
				# add_dynasty_prestige_level = -1
			# }
			remove_global_variable = coa_new_name
			set_variable = {
				name = my_coa_belongs_to
				value = scope:actor
			}
		}
		scope:actor = {
			set_variable = {
				name = my_personal_coa
				value = scope:recipient
			}
			set_relation_agot_personal_coa_slave = scope:recipient
			scope:recipient = { save_scope_as = exiled_dummy } # Send away the COA Dummy Immediately since the AI won't be editing their own COAs
			trigger_event = agot_events_personal_coat_of_arms.0001
		}
	}
}
