asoiaf_equip_aegon_I_crown_artefact_effect = {
	every_character_artifact = {
		limit = { has_variable = crown_aegon_i_artifact }
		save_scope_as = this_artifact
	}
	scope:this_artifact = {
		equip_artifact_to_owner_replace = yes
	}
}
