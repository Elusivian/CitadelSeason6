agot_assign_suspicious_strong_seed_effect = {
	if = {
		limit = {
			exists = father
			NOT = { father = scope:real_father }
			NOT = { father.dynasty = scope:real_father.dynasty }
			NOT = { scope:mother.dynasty = scope:real_father.dynasty }
		}
		add_character_flag = suspicious_strong_seed
	}
}

					real_father.real_father ?= { strong_seed_height_matches_dynasty_trigger = yes }
				}
			}
		}
		if = {
			limit = {
				AND = {
					real_father = { has_strong_seed_height_trigger = yes }
					real_father = { strong_seed_height_matches_dynasty_trigger = yes }
				}
			}
			agot_assign_strong_seed_height_traits_effect = { GENE_DONOR = real_father }
			agot_assign_suspicious_strong_seed_effect = yes
		}
		else_if = {
			limit = {
				AND = {
					real_father.mother.dynasty ?= dynasty
					real_father.mother ?= { has_strong_seed_height_trigger = yes }
					real_father.mother ?= { strong_seed_height_matches_dynasty_trigger = yes }
				}
			}
			agot_assign_strong_seed_height_traits_effect = { GENE_DONOR = real_father.mother }
		}
		else_if = {
			limit = {
				AND = {
					real_father.real_father.dynasty ?= dynasty
					real_father.real_father ?= { has_strong_seed_height_trigger = yes }
					real_father.real_father ?= { strong_seed_height_matches_dynasty_trigger = yes }
				}
			}
			agot_assign_strong_seed_height_traits_effect = { GENE_DONOR = real_father.real_father }
		}
	}
	else_if = {
		limit = {
			AND = {
				scope:real_father = { has_strong_seed_eye_lid_trigger = yes }
				scope:mother = { has_strong_seed_eye_lid_trigger = yes }
			}
		}
		random_list = {
			60 = {
				modifier = {
					add = 10
					scope:real_father.dynasty.dynasty_prestige_level > scope:mother.dynasty.dynasty_prestige_level
				}
				modifier = {
					add = -10
					scope:real_father.dynasty.dynasty_prestige_level < scope:mother.dynasty.dynasty_prestige_level
				}
				modifier = {
					add = -40
					scope:real_father = { strong_seed_eye_lid_match_dynasty_trigger = no }
				}
				agot_assign_strong_seed_eye_lid_traits_effect = { GENE_DONOR = scope:real_father }
				agot_assign_suspicious_strong_seed_effect = yes
			}
			40 = {
				modifier = {
					add = -10
					scope:real_father.dynasty.dynasty_prestige_level > scope:mother.dynasty.dynasty_prestige_level
				}
				modifier = {
					add = 10
					scope:real_father.dynasty.dynasty_prestige_level < scope:mother.dynasty.dynasty_prestige_level
				}
				modifier = {
					add = -26
					scope:mother = { strong_seed_eye_lid_match_dynasty_trigger = no }
				}
				#base chance 60/40 in favor of mother for matrimarriage
				modifier = {
					add = 50
					exists = dynasty
					dynasty ?= mother.dynasty
					scope:mother = { strong_seed_eye_lid_match_dynasty_trigger = yes }
				}
				agot_assign_strong_seed_eye_lid_traits_effect = { GENE_DONOR = scope:mother }
			}
		}
	}
	else_if = { #Only father has seed
		limit = {
			scope:real_father = { has_strong_seed_eye_lid_trigger = yes }
		}
		random_list = {
			34 = { #Seed stays strong
				agot_assign_strong_seed_eye_lid_traits_effect = { GENE_DONOR = scope:real_father }
				agot_assign_suspicious_strong_seed_effect = yes
			}
			66 = { #No seed
				modifier = {
					add = -66
					scope:real_father = { strong_seed_eye_lid_match_dynasty_trigger = yes }
				}
			}
		}
	}
	else_if = { # Only mother has seed
		limit = {
			scope:mother = { has_strong_seed_eye_lid_trigger = yes }
		}
		random_list = {
			34 = { #Seed stays strong
				agot_assign_strong_seed_eye_lid_traits_effect = { GENE_DONOR = scope:mother }
			}
			66 = { #No seed
				modifier = {
					add = -66
					scope:mother = { strong_seed_eye_lid_match_dynasty_trigger = yes }
				}
			}
		}
	}

	if = {
		limit = { has_strong_seed_eye_lid_trigger = no }
		random_list = {
			1 = {
				trigger = {
					scope:mother.mother ?= { has_strong_seed_eye_lid_trigger = yes }
				}
				modifier = {
					dynasty ?= scope:mother.mother.dynasty
					scope:mother.mother = { strong_seed_eye_lid_match_dynasty_trigger = yes }
					factor = 4
				}
				agot_assign_strong_seed_eye_lid_traits_effect = { GENE_DONOR = scope:mother.mother }
			}
			1 = {
				trigger = {
					scope:mother.real_father ?= { has_strong_seed_eye_lid_trigger = yes }
				}
				modifier = {
					dynasty ?= scope:mother.real_father.dynasty
					scope:mother.real_father = { strong_seed_eye_lid_match_dynasty_trigger = yes }
					factor = 4
				}
				agot_assign_strong_seed_eye_lid_traits_effect = { GENE_DONOR = scope:mother.real_father }
			}
			1 = {
				trigger = {
					scope:real_father.mother ?= { has_strong_seed_eye_lid_trigger = yes }
				}
				modifier = {
					dynasty ?= scope:real_father.mother.dynasty
					scope:real_father.mother = { strong_seed_eye_lid_match_dynasty_trigger = yes }
					factor = 4
				}
				agot_assign_strong_seed_eye_lid_traits_effect = { GENE_DONOR = scope:real_father.mother }
			}
			1 = {
				trigger = {
					scope:real_father.real_father ?= { has_strong_seed_eye_lid_trigger = yes }
				}
				modifier = {
					dynasty ?= scope:real_father.real_father.dynasty
					scope:real_father.real_father = { strong_seed_eye_lid_match_dynasty_trigger = yes }
					factor = 4
				}
				agot_assign_strong_seed_eye_lid_traits_effect = { GENE_DONOR = scope:real_father.real_father }
			}
			4 = { }
		}
	}
}
