#Sets the correct "parent house" for a known bastard
agot_set_parent_house_effect = {
	if = {
		limit = {
			should_mother_give_house_to_bastard_trigger = yes
		}
		set_house = mother.house
	}
	else_if = {
		limit = {
			exists = father
			exists = father.house
			NOT = { father = { has_trait = bastard } }
		}
		set_house = father.house
	}
	else = {
		set_to_lowborn = yes
	}
}
