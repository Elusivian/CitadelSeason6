### GoT mod - Set Patron Effect
### Used for selecting a patron god
agot_set_patron_effect = {
	if = {
		limit = {
			NOT = {
				has_character_modifier = patron_$PATRON$_modifier
			}
		}

		agot_clear_patron = yes

		add_piety = minor_piety_loss

		set_variable = {
			name = patron_god
			value = flag:$PATRON$
		}
		add_character_modifier = patron_$PATRON$_modifier
	}
}

agot_clear_patron = {
	every_patron_name_complex = { APPLY = agot_remove_patron_modifier }

	remove_variable = patron_god
}

agot_remove_patron_modifier = {
	remove_character_modifier = patron_$PATRON$_modifier
}
