#Scripted Modifiers used for Activities
base_accept_values_modifier = {
	modifier = {
		add = 30
	}
	modifier = {
		add = 30
		scope:recipient = {
			has_relation_friend = scope:actor
		}
	}
	modifier = {
		add = 60
		scope:recipient = {
			has_relation_best_friend = scope:actor
		}
	}
	opinion_modifier = {
		who = scope:recipient
		opinion_target = scope:actor
		multiplier = 0.5
	}
}

base_ignore_values_modifier = {
	modifier = {
		add = 60
	}
	modifier = {
		add = 40
		scope:recipient = {
			NOT = { is_close_family_of = scope:actor }
		}
	}
}

base_reject_values_modifier = {
	modifier = {
		add = 60
	}
	opinion_modifier = {
		who = scope:recipient
		opinion_target = scope:actor
		multiplier = -1
	}
	modifier = {
		add = 60
		scope:recipient = {
			has_relation_rival = scope:actor
		}
	}
}
