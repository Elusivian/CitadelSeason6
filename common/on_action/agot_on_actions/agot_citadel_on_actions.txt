on_game_start = {
	on_actions = {
		on_citadel_start
	}
}

on_citadel_start = {
	effect = {
		if = {
			limit = {
				exists = global_var:citadel_title
			}
			global_var:citadel_title.holder = {
				change_government = the_citadel_government
			}
		}
		else = {
			title:b_the_citadel.holder = {
				change_government = the_citadel_government
			}
		}
	}
}
