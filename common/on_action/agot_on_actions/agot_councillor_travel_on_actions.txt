# Pause councillor travel indefinitely while the councillor is performing their task
agot_on_arrival_councillor_travel = {
	trigger = {
		current_travel_plan ?= {
			has_variable = agot_councillor_travel
			is_cancelled = no
		}
	}
	effect = {
		current_travel_plan = {
			pause_travel_plan = yes
		}
	}
}
