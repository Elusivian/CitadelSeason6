personal_steward_default_task_diplomacy_scale = {
	add = diplomacy
	if = {
		limit = {
			scope:councillor_liege = { has_perk = loyalty_and_respect_perk }
		}
		multiply = 1.25
	}
	if = {
		limit = {
			scope:councillor_liege.culture = { has_cultural_parameter = better_spouses }
		}
		multiply = 1.25
	}
	divide = 5
	if = {
		limit = {
			OR = {
				scope:councillor_liege = { has_perk = loyalty_and_respect_perk }
				scope:councillor_liege.culture = { has_cultural_parameter = better_spouses }
			}
		}
		ceiling = yes
	}
	else = {
		floor = yes
	}
	min = 1
}
