agot_tywin_join_side_chance_mult = 0.1

agot_get_attacker_war_score_value = {
	add = var:usurper_war.attacker_war_score
	multiply = agot_tywin_join_side_chance_mult

	if = {
		limit = { character:Targaryen_1 = { is_alive = yes } }

		multiply = 0.5
	}
}

agot_get_defender_war_score_value = {
	add = var:usurper_war.defender_war_score
	multiply = agot_tywin_join_side_chance_mult

	if = {
		limit = { character:Targaryen_1 = { is_alive = yes } }

		multiply = 1.5
	}
}
