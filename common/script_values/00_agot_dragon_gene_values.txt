# Each svalue (scripted value) is split into two IF statements, one for the Dragonrider poses (identifed by flag currently riding)
# the other If statement is for retaining dragon appearances when they die
# If new dragon genes are added, you have to add the appropriate svalue in the format everything else uses below
# You then call the 'gene_name_svalue' in the portrait modifier value slot!

dragon_age_svalue = {
	value = {
		if = {
			limit = {
				has_character_flag = currently_riding_dragon
				agot_ruins_has_dragon_storage_system = yes
			}
			title:c_ruins.holder = {
				every_owned_story = {
					limit = {
						story_type = story_dragon_variable_storage
						var:dragon_id ?= root.var:current_dragon
					}
					save_temporary_scope_as = dragon_var_story_val
				}
			}
			value = scope:dragon_var_story_val.var:dragon_age
		}
		if = {
			limit = {
				agot_ruins_has_dragon_storage_system = yes
			}
			title:c_ruins.holder = {
				every_owned_story = {
					limit = {
						story_type = story_dragon_variable_storage
						var:dragon_id ?= root
					}
					save_temporary_scope_as = dragon_var_story_val
				}
			}
			value = scope:dragon_var_story_val.var:dragon_age
		}
	}
}
