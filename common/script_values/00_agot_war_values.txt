slave_raid_ai_score_value = {
	value = 0
	scope:attacker = {
		if = {
			limit = {
				ai_greed > 0
			}
			add = {
				value = ai_greed
				divide = 2
			}
		}
		if = {
			limit = {
				ai_compassion < 0
			}
			add = {
				value = ai_compassion
				multiply = -1
				divide = 2
			}
		}
		if = {
			limit = {
				NOT = { agot_can_upgrade_any_slave_population_trigger = yes }
			}
			multiply = 0
		}
	}
}

free_slaves_ai_score_value = {
	value = 0
	scope:attacker = {
		if = {
			limit = {
				ai_zeal >= very_high_positive_ai_value
				OR = {
					religion = religion:faith_of_the_seven_religion
					religion = religion:old_gods_religion
				}
			}
			add = 5
		}
		if = {
			limit = {
				ai_compassion >= very_high_positive_ai_value
			}
			add = 5
		}
		if = {
			limit = {
				ai_greed >= medium_positive_ai_value
			}
			multiply = 0
		}
		if = {
			limit = {
				NOT = { has_realm_law = crown_authority_3 }
			}
			multiply = 0
		}
	}
}

nw_generic_ranging_ai_score_mult_value = {
	value = 1

	# Weigh up developed counties
	every_in_list = {
		list = target_titles
		limit = { development_level >= 1 }
		add = 0.25
	}

	# Weigh up counties with more buildings
	every_in_list = {
		list = target_titles
		limit = {
			any_county_province = { num_buildings >= 2 }
		}
		add = 0.25
	}

	# Weigh up defenders who are already at war
	if = {
		limit = {
			scope:defender = { is_at_war = yes }
		}
		add = 0.25
	}

	min = 0
}

stepstones_adventurer_target_title_priority_value = {
	value = 0
	#Mark up larger sub-realms
	add = {
		value = 5
		multiply = holder.domain_size
	}
	#Mark up coastal counties.
	if = {
		limit = { is_coastal_county = yes }
		add = 10
	}
	#Mark up potential rivals & rivals.
	if = {
		limit = {
			holder = {
				any_relation = {
					type = potential_rival
					this = scope:stepstones_adventurer
				}
			}
		}
		add = 25
	}
	if = {
		limit = {
			scope:stepstones_adventurer = {
				any_relation = {
					type = rival
					this = scope:stepstones_adventurer
				}
			}
		}
		add = 50
	}
	#Mark down potential friends & friends.
	if = {
		limit = {
			scope:stepstones_adventurer = {
				any_relation = {
					type = potential_friend
					this = scope:stepstones_adventurer
				}
			}
		}
		add = -25
	}
	if = {
		limit = {
			scope:stepstones_adventurer = {
				any_relation = {
					type = friend
					this = scope:stepstones_adventurer
				}
			}
		}
		add = -50
	}
	#Mark down potential lovers & lovers.
	if = {
		limit = {
			scope:stepstones_adventurer = {
				any_relation = {
					type = potential_lover
					this = scope:stepstones_adventurer
				}
			}
		}
		add = -25
	}
	if = {
		limit = {
			scope:stepstones_adventurer = {
				any_relation = {
					type = lover
					this = scope:stepstones_adventurer
				}
			}
		}
		add = -50
	}
	if = {
		limit = {
			holder = {
				NOT = { has_government = pirate_government }
			}
		}
		add = 50
	}
	#Mark down players a little more, since we don't want them to be constantly fending off pagans unless they've got the juciest lands around.
	if = {
		limit = {
			holder = { is_ai = no }
		}
		add = -20
	}
	#Mark down if holder has same faith.
	if = {
		limit = { holder.faith = scope:stepstones_adventurer.faith }
		add = -10
	}
	if = {
		limit = { holder.culture = scope:stepstones_adventurer.culture }
		add = -10
	}
}
