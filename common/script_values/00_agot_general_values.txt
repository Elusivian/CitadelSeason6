# Faith hostility level between the faith of the scoped character
# and the faith of scope:target_character, 0-3.
agot_faith_hostility_level_value = {
	if = {
		limit = {
			faith = {
				faith_hostility_level = {
					target = scope:target_character.faith
					value < faith_astray_level
				}
			}
		}
		value = 0
	}
	else_if = {
		limit = {
			faith = {
				faith_hostility_level = {
					target = scope:target_character.faith
					value < faith_hostile_level
				}
			}
		}
		value = 1
	}
	else_if = {
		limit = {
			faith = {
				faith_hostility_level = {
					target = scope:target_character.faith
					value < faith_evil_level
				}
			}
		}
		value = 2
	}
	else = {
		value = 3
	}
}

# Dread level of the scoped character towards scope:target_character, 0-2.
agot_dread_level_value = {
	if = {
		limit = {
			has_dread_level_towards = {
				target = scope:target_character
				level = 2
			}
		}
		value = 2
	}
	else_if = {
		limit = {
			has_dread_level_towards = {
				target = scope:target_character
				level = 1
			}
		}
		value = 1
	}
	else = {
		value = 0
	}
}

# Scoped character's opinion of scope:target_character, -100-100.
agot_opinion_value = {
	value = 0
	if = {
		limit = {
			save_temporary_opinion_value_as = {
				name = opinion_of_target_character
				target = scope:target_character
			}
		}
		value = scope:opinion_of_target_character
	}
}

# Average powerful vassals opinion of the scoped character, -100-100.
agot_powerful_vassals_opinion_value = {
	value = 0
	every_powerful_vassal = {
		if = {
			limit = {
				save_temporary_opinion_value_as = {
					name = vassal_opinion
					target = prev
				}
			}
			add = scope:vassal_opinion
		}
	}
	if = {
		limit = { number_of_powerful_vassals > 0 }
		divide = number_of_powerful_vassals
	}
}
