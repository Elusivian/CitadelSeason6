at_custom_artifact_base_price_value = {
	value = 0
	if = {
		limit = { agot_is_valyrian_steel_artifact = yes }
		add = {
			value = 1000
			desc = agot_artifact_price_valyrian_steel_desc
		}
	}
	if = {
		limit = { agot_is_dragon_egg_artifact = yes }
		add = {
			value = 1000
			desc = agot_artifact_price_dragon_egg_desc
		}
	}
}

at_custom_artifact_bid_price_multiplier = {
	value = 1
	if = {
		limit = { agot_is_dragon_egg_artifact = yes }
		multiply = scope:buyer.agot_dragon_egg_bid_price_multiplier
	}
	if = {
		limit = {
			has_variable = agot_new_historical_crown_by_decision
		}
		multiply = {
			value = 0.3
			desc = agot_new_historical_crown_by_decision_desc
		}
	}
}
