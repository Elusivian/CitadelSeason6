agot_artifact_bid_price_value = {
	global_var:agot_artifact_keeper = { save_temporary_scope_as = buyer }
	value = at_artifact_bid_price_value
}

agot_artifact_ask_price_value = {
	global_var:agot_artifact_keeper = { save_temporary_scope_as = seller }
	value = at_artifact_ask_price_value
}

agot_dragon_eggs_count_value = {
	value = 0
	every_character_artifact = {
		limit = { agot_is_dragon_egg_artifact = yes }
		add = 1
	}
}

agot_dragon_egg_bid_price_multiplier = {
	if = {
		limit = {
			OR = {
				NOT = { has_variable = agot_dragon_eggs_count }
				var:agot_dragon_eggs_count = 0
			}
		}
		value = 1
	}
	else_if = {
		limit = { var:agot_dragon_eggs_count = 1 }
		value = 0.7
	}
	else_if = {
		limit = { var:agot_dragon_eggs_count = 2 }
		value = 0.5
	}
	else_if = {
		limit = { var:agot_dragon_eggs_count = 3 }
		value = 0.15
	}
	else = {
		value = 0
	}
}
