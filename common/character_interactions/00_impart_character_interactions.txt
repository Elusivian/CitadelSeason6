impart_values_interaction = {
	category = interaction_category_friendly
	common_interaction = yes
	icon = child
	desc = impart_values_interaction_desc

	ai_frequency = 0

	cooldown_against_recipient = { years = 3 }

	is_shown = {
		has_bp2_dlc_trigger = no
		NOT = { scope:actor = scope:recipient }
		scope:actor = { 
			is_adult = yes
			is_ai = no
		}
		scope:recipient = { 
			is_adult = no
			age > 9
			number_of_personality_traits < personality_trait_limit
		}
		
		scope:recipient = { 
		    OR = {
				has_relation_ward = scope:actor
				is_close_family_of = scope:actor
				AND = {
					exists = house
					scope:actor.house = house
					scope:actor = house.house_head
				}
			}
		}
		scope:actor = { is_human = yes }
		scope:recipient = { is_human = yes }
	}
		
	is_valid_showing_failures_only = {
		scope:actor = {
			NOT = { is_at_war_with = scope:recipient }
		}
		scope:recipient = {
			custom_description = {
				text = already_imparted_lesson_desc
				object = scope:recipient
				NOT = {
					has_character_flag = imparted_value_lesson_taught
				}
			}
		}
	}
	
	auto_accept = yes
	
	on_accept = {
		scope:actor = {
			trigger_event = impart_values.0001
		}
	}
}
