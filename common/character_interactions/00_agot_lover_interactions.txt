become_paramours_interaction = {
	category = interaction_category_friendly
	use_diplomatic_range = yes
	icon = lover

	desc = become_paramours_interaction_desc

	is_shown = {
		scope:actor = {
			has_relation_lover = scope:recipient
		}
		OR = {
			AND = {
				scope:recipient = { agot_can_add_paramour = yes }
				scope:actor = { agot_can_be_paramour = yes }
			}
			AND = {
				scope:actor = { agot_can_add_paramour = yes }
				scope:recipient = { agot_can_be_paramour = yes }
			}
		}
	}
	on_accept = {
		if = {
			limit = {
				scope:actor = { agot_can_have_paramour = yes }
			}
			scope:actor = { save_scope_as = noble }
			scope:recipient = { save_scope_as = paramour }
		}
		else = {
			scope:recipient = { save_scope_as = noble }
			scope:actor = { save_scope_as = paramour }
		}
		agot_become_paramour_effect = yes
	}
	auto_accept = yes
}
